# 题目分析
- 升序排列的数组 A
    - 二分
- 在 A 中找与 target 最接近的 k 个整数
    - 先找到 target 从中间向两边扩展
    - 无 target 找到最接近的数作为中界

# 代码题解
```py
class Solution:
    """
    @param a: an integer array
    @param target: An integer
    @param k: An integer
    @return: an integer array
    """
    def k_closest_numbers(self, a, target, k):
        # write your code here
        # 找到最左边 >= target 的数
        right = self.findUpperCloest(a, target)
        left = right - 1
        # 两指针从中间扩散找 k 个数
        # 保存到数组
        result = []

        for _ in range(k):
            # 左右哪个更接近
            if self.isLeftClosest(a, target, left, right):
                result.append(a[left])
                left -= 1
            else:
                result.append(a[right])
                right += 1
        
        return result

    def findUpperCloest(self, a, target):
        start, end = 0, len(a) - 1

        while start + 1 < end:
            mid = (start + end) / 2

            if a[mid] <= target:
                start = mid
            else:
                end = mid
        
        if a[start] >= target:
            return start
        if a[end] >= target:
            return end
        return len(a)
    
    def isLeftClosest(self, a, target, left, right):
        if left < 0:
            return False
        if right >= len(a):
            return True
        
        # 优先选择左边
        return target - a[left] <= a[right] - target

```
